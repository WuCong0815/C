# include <stdio.h>

int main(void)
{

	int n = 0;
	int m = 1;
        int j = 0;
	int	h;
	int	k;

	printf("请输入您想查找的斐波那契数的位置数： ");
	scanf("%d",&k);

	if(1 == k)
		j = 0;
	
	else if(2 == k)
		j = 1;

	else
		for(h = 2;h < k;h++)
		{
			j = n + m;
			n = m;
			m = j;
		}

        printf("%d\n",j);
	return 0;


}
