# include <stdio.h>
# include <math.h>
int main(void)
{    
	double a;
	double b;
	double c;
	double delta;
	double x1;
	double x2;
        
     printf("请依次输入一元二次方程的三个参数：\n");
     scanf("%f %f %f\n",&a, &b, &c);

	delta = b*b - 4*a*c;

if (delta > 0)
	{  
		x1 = (-b - sqrt(delta)) / (2*a);
		x2 = (-b + sqrt(delta)) / (2*a);
		printf("此方程有两个解:x1 = %f,x2 = %f\n",x1,x2);

	}
  else if (delta == 0)
{
 	 x1 = (-b) / (2*a);
	 x2 = x1;
	 printf("此方程有一个解:x1 = x2=%f\n",x1);
 }

else 
{
  printf("此方程无解\n");
}    

    return 0;
}
